program
    var NULL as int;
    var A as int;
    var B as int;
    var TEMP as int;
begin
    NULL := readInt;
    if NULL > 0 then
        A := 0;
        B := 1;
        while NULL > 0 do
            writeInt A;
            TEMP := A + B;
            A := B;
            B := TEMP;
            NULL := NULL - 1;
        end;
    end;
end
